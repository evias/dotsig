24 int main(int argc, char* argv[])
25 {
26  // fills dotsig::OPTIONS

29  // rapidly determine if the call contains -h or -v
30  if (dotsig::get_flag("-h")) return dotsig::print_usage();
31  else if (dotsig::get_flag("-v")) return dotsig::print_version();
32 
33  // registers supported identity types
34  dotsig::Factory* FACTORY = new dotsig::Factory();
35  dotsig::InitializeFactory(FACTORY);
36 
37  // parses possible file and -a options
38  std::vector FILES = dotsig::get_files();
39  std::string file = dotsig::get_option("file"),
40  algo = dotsig::get_option("-a"),
41  mode = dotsig::get_flag("-c") ? "Verification" : "Signature",
42  priv = dotsig::get_option("-i"),
43  pub = dotsig::get_option("-P"),
44  buffer;
45 
46  // accepts data on stdin (e.g. `cat data/document | dotsig`)
47  if (file.empty() || (FILES.size() == 1 && file.ends_with(".sig"))) {
48  buffer = dotsig::consume_stdin();
49  dotsig::OPTIONS.emplace("stdin", buffer);
50  }
51 
52  // at least one file or stdin input are required
53  if (file.empty() && buffer.empty()) {
54  return dotsig::print_usage();
55  }
56 
57  // passphrase input with echo suppressed
58  // note: use Ctrl+D to stop input on Unix, Ctrl+Z on Windows
59  std::string pass = dotsig::get_option("-p");
60  if (pass.empty() || pass == "-") pass = dotsig::get_password();
61 
62  // accepts "ecdsa" (default), "pkcs", "openpgp", "openpgp:rsa", etc.
63  algo = dotsig::get_dsa_type(algo);
64 
65  debug() << "Algorithm: " << algo << std::endl
66  << "Mode: " << mode << std::endl;
67 
68  std::string id_file, message;
69  try {
70  // creates a IIdentity subclass object by algorithm
71  auto identity = FACTORY->MakeIdentity(algo);
72 
73  // in signature mode:
74  // accepts "-i" identity file or defaults to ~/id_ecdsa
75  // note: this is platform-dependent and uses APPDATA on Windows.
76  if (! dotsig::get_flag("-c")) {
77  id_file = priv.empty() ? dotsig::get_identity_file(algo) : priv;
78  }
79  // in verification mode:
80  // accepts "-P" public key or defaults to ~/id_ecdsa.pub
81  // note: this is platform-dependent and uses APPDATA on Windows.
82  else {
83  id_file = pub.empty() ? dotsig::get_public_identity_file(algo) : pub;
84  }
85 
86  std::filesystem::directory_entry entry{id_file};
87 
88  debug() << "Using identity file: "
89  << id_file
90  << (entry.exists() ? " (load)" : " (new)")
91  << std::endl;
92 
93  // loads an identity from file (DER for private keys, PEM for public keys)
94  if (entry.exists()) {
95  identity->Import(id_file, pass);
96  }
97  // or creates a new identity and exports to file
98  else {
99  identity->GenerateRandom();
100  identity->Export(id_file, pass);
101  }
102 
103  // consumes messages/signatures to sign/verify from file(s)
104  std::map<std::string, std::string> messages{};
105  messages = dotsig::consume_inputs(FILES);
106 
107  // consumes original message from stdin (if available)
108  if (! buffer.empty()) messages.insert(std::make_pair("stdin", buffer));
109 
110  // iterate through processed <file> options
111  // in signature mode: sign the processed data directly.
112  // in verification mode: find the corresponding file, then verify.
113  for (auto msg = messages.begin(); msg != messages.end(); ++msg) {
114  std::string current = (*msg).first;
115 
116  // in signature mode:
117  if (! dotsig::get_flag("-c")) {
118  // signs input files and stores signatures in colocated .sig file(s)
119  auto signature = identity->Sign(
120  (*msg).second,
121  current + ".sig"
122  );
123 
124  std::cout << "Signature: " << signature << std::endl;
125  continue;
126  }
127 
128  // in verification mode:
129  // skip non-dotsig files, used only to forward verifiable content
130  if (! current.ends_with(".sig")) continue;
131 
132  // prepare inputs discovery for original message
133  std::string doc_file = current.substr(0, current.find(".sig")), doc_buffer;
134  auto input_it = messages.find(doc_file);
135  auto stdin_it = messages.find("stdin");
136 
137  // find document (original message) from inputs
138  if (input_it != messages.end()) {
139  doc_buffer = (*input_it).second;
140  }
141  // find document (original message) from stdin
142  else if (stdin_it != messages.end()) {
143  doc_buffer = (*stdin_it).second;
144  }
145  // dotsig *must* know the original message
146  else throw std::runtime_error(
147  "Missing document to verify signature: " + current
148  );
149 
150  // verify signature x for original message
151  auto result = identity->Verify((*msg).second, doc_buffer);
152  std::cout << "Verified " << (*msg).first << ": "
153  << (result ? "OK" : "NOT OK")
154  << std::endl;
155  }
156 
157  delete identity;
158  delete FACTORY;
159  }
160  catch (std::runtime_error& e) {
161  std::cerr << "An error ocurred: " << e.what() << std::endl;
162  return 1;
163  }
164 
165  return 0;
166 }
