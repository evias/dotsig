25  std::vector<std::string> inputs
26 ) {

28 
29  for (auto it = inputs.begin(); it != inputs.end(); ++it) {
30  std::filesystem::directory_entry file{*it};
31  if (! file.exists())
32  throw std::runtime_error("Error: Provided document does not exist: " + (*it));
33 
34  // reads the content from file

38  file_ptr.close();

41  messages.insert(std::make_pair(*it, file_buf.str()));
42  }
43

45 }
