53  /// \brief Template class for OpenPGP identities that consist of a private/public
54  /// keypair which may be generated using one of: RSA, DSA, ECDSA and EdDSA.
55  ///
56  /// This template class accepts three template parameters with PrivateKeyImpl,
57  /// PublicKeyImpl and SubKeyImpl. Importantly, the current implementation of
58  /// the copy constructor makes it so that these template parameters must be
59  /// compatible with Botan's pk_keys.h format. Method implementations must be
60  /// provided by these classes for the methods: algorithm_identifier, private_key_bits
61  /// and public_key_bits, as found in Botan::Private_Key and Botan::Public_Key.
62  ///
63  /// \note The OpenPGP standard defines compatibility with multiple digital
64  /// signature algorithms including: RSA (PKCS1 v1.5) [default], DSA, ECDSA, EdDSA.
65  ///
66  /// \note Also, the OpenPGP standard is different from basic ECDSA and PKCS
67  /// implementations because it uses a concept of armored messages and keys. Yet,
68  /// the actual raw signatures produced using OpenPGP identities *do not differ*
69  /// from signatures created using the PKCS::Identity and ECDSA::Identity
70  /// classes.
71  ///
72  /// \tparam PrivateKeyImpl The private key implementation class.
73  /// \tparam PublicKeyImpl The public key implementation class.
74  /// \tparam SubKeyImpl The sub (child) key implementation class.
75  ///
76  /// \todo Implement sub-keys as defined by the OpenPGP standard.
77  /// \todo Implement expiration logic and signature/keys revocation strategies.
78  /// \todo Diverge from original PKA implementations by usage of **subkeys**.
79  /// \see dotsig::IIdentity
80  template <
81  class PrivateKeyImpl,
82  class PublicKeyImpl,
83  class SubKeyImpl = PublicKeyImpl
84  >
85  class Identity
86  : public IIdentity
87  {
88  protected:
89  /// \brief Contains a unique pointer to the private key implementation.
90  /// \note This member variable is wiped-out ("zero'd") by the destructor.
91  std::unique_ptr<PrivateKeyImpl> m_private_key;
92 
93  public:
94  /// \brief Contains a unique pointer to the public key implementation.
95  /// \note This member variable is wiped-out ("zero'd") by the destructor.
96  std::unique_ptr<PublicKeyImpl> m_public_key;
97 
98  /// \brief Contains a *padding scheme with hash function* in the case of
99  /// OpenPGP with RSA (PKCS1 v1.5), otherwise contains a *hash function*.
100  std::string m_scheme;
101 
102  /// \brief Default constructor. Creates an empty OpenPGP identity.
103  ///
104  /// This method accepts a string-typed \a scheme that further defines the
105  /// algorithm used during signatures creation and verification. A parameter
106  /// is necessary here because OpenPGP provides multiple compatible DSA types.
107  ///
108  /// \note The created identity does not have a private key, make sure to
109  /// use one of GenerateRandom or Import to populate it.
110  ///
111  /// \param scheme The padding scheme with hash function OR only a hash function.
112  /// \see m_scheme
113  /// \see GenerateRandom
114  /// \see Import
115  /// \see Export
116  Identity(const std::string& scheme = "PKCS1v15(SHA-256)")
117  : IIdentity(), m_scheme(scheme)/*, m_sub_keys({})*/ {}
118 
119  /// \brief Copy constructor. Creates an identity based on the other's private key.
120  Identity(const Identity&);
121 
122  /// \brief Default class destructor.
123  virtual ~Identity() = default;
124 
125  /// \brief Creates an identity with file \a filename and pass \a passphrase.
126  /// \param filename The complete filesystem path to an encrypted identity file.
127  /// \param passphrase A passphrase to decrypt the encrypted identity file.
128  /// \return *this
129  /// \see GenerateRandom
130  /// \see Export
131  const Identity<PrivateKeyImpl, PublicKeyImpl, SubKeyImpl>& Import(
132  const std::string&,
133  const std::string&
134  ) override;
135 
136  /// \brief Saves an identity to file \a filename with password \a passphrase.
137  /// \note A public key file will also be created at {filename}.pub.
138  /// \param filename The filesystem path where the identity file will be stored.
139  /// \param passphrase A passphrase to encrypt the identity file.
140  /// \see GenerateRandom
141  /// \see Import
142  void Export(const std::string&, const std::string&) const override;
143 
144  /// \brief Signs a message \a message and saves the signature to \a sig_file.
145  /// \param message The complete message for which a digital signature is created.
146  /// \param sig_file The filesystem path where the signature file will be stored.
147  /// \see Verify
148  std::string Sign(const std::string&, const std::string&) const override;
149 
150  /// \brief Verifies a signature \a signature for a message \a message.
151  /// \param signature The signature bytes, as stored in a signature file (not hex!).
152  /// \param message The complete message for which a digital signature is verified.
153  /// \see Sign
154  bool Verify(const std::string&, const std::string&) const override;
155 
156  /// \brief Generates a random pair of private- and public-key.
157  virtual void GenerateRandom() override = 0;
158  };

263  /// the signature scheme, instead of using the pre-hashed variants which are
264  /// recommended by Botan. Using the GnuPG variant is obviously for compatiblity
265  /// with the OpenPGP standard in this namespace.
266  ///
267  /// \note Signatures are **not compatible** with `Ed25519ph` signature scheme.
268  /// \note This identity wrapper exports BER-encoded private keys to the user's

278  /// parameters specification being: `SHA-512`. This implies the use of a
279  /// `GnuPG`-compatible Ed25519 variant of the signature scheme, instead of
280  /// using the pre-hashed variants which are recommended. Using the GnuPG
281  /// variant is obviously for compatiblity with the OpenPGP standard.
282  ///
283  /// \note Signatures are **not compatible** with `Ed25519ph` signature scheme.
