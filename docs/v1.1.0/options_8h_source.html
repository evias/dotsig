23  /// \brief Parses the command options using \a argc and \a argv.
24  /// \param argc Contains the number of options passed to the program.
25  /// \param argv Contains the option values as passed to the program.
26  inline void parse_args(int argc, char* argv[]) {
27  std::vector flags = {"-v", "-h", "-c", "-D", "-q"};
28  for (int i = 0; i < argc; ++i) {
29  std::string opt(argv[i]);
30  if (i == 0) OPTIONS.emplace("program", opt);
31  // options and flags are prefixed with "-"
32  // @todo: does this *have to be* "/" in Windows?
33  else if (argv[i][0] == '-' && opt.size() > 1) {
34  std::string nxt;
35  bool is_flag = flags.end() != std::find(flags.begin(), flags.end(), opt);
36 
37  // value options, e.g.: `-p passphrase`
38  if (!is_flag && opt.size() == 2 && argc >= i+2 && (
39  (nxt = argv[i+1])[0] != '-' || nxt == "-"
40  )) {
41  OPTIONS.emplace(opt, std::string(argv[i+1]));
42  i++; // force skip value
43  }
44  // flags, e.g.: `-cq`
45  else {
46  for (int j = 1, n = opt.size(); j < n; ++j) {
47  std::string flag("-"); flag.push_back(argv[i][j]);
48  OPTIONS.emplace(flag, "1");
49  }
50  }
51  }
52  else if (OPTIONS.end() == OPTIONS.find("file")) {
53  OPTIONS.emplace("file", opt);
54  }
55  else /* <fileN> */ {
56  for (int h = 2; ; ++h) {
57  std::string file_n("file"); file_n.append(std::to_string(h));
58  if (OPTIONS.end() == OPTIONS.find(file_n)) {
59  OPTIONS.emplace(file_n, opt);
60  break;
61  }
62  }
63  }
64  }
65  }
