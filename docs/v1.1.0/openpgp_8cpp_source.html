51 const dotsig::OpenPGP::Identity<PrivateKeyImpl, PublicKeyImpl, SubKeyImpl>&
52 dotsig::OpenPGP::Identity<PrivateKeyImpl, PublicKeyImpl, SubKeyImpl>::Import(
53  const std::string& filename,
54  const std::string& passphrase
55 ) {
56  std::string err_string;
57 
58  // try to import an unencrypted public-key (from: id_rsa.pub, id_ecdsa.pub)
59  try {
60  Botan::DataSource_Stream input(filename); // non-binary mode (PEM)
61  std::unique_ptr<Botan::Public_Key> pub = Botan::X509::load_key(input);
62 
63  m_public_key = std::make_unique<PublicKeyImpl>(
64  pub->algorithm_identifier(),
65  pub->public_key_bits()
66  );
67 
68  return *this;
69  }
70  catch(Botan::Exception& e) {}
71 
72  // or try to load using an encrypted private key (from: id_rsa, id_ecdsa)
73  try {
74  Botan::DataSource_Stream input(filename, true); // binary mode (BER)
75  std::unique_ptr<Botan::Private_Key> priv = Botan::PKCS8::load_key(
76  input,
77  passphrase
78  );
79 
80  m_private_key = std::make_unique<PrivateKeyImpl>(
81  priv->algorithm_identifier(), priv->private_key_bits()
82  );
83 
84  m_public_key = std::make_unique<PublicKeyImpl>(
85  m_private_key->algorithm_identifier(),
86  m_private_key->public_key_bits()
87  );
88 
89  return *this;
90  }
91  catch(Botan::Exception& e) {
92  err_string = e.what();
93  }
94 
95  // private key could not be loaded
96  throw std::runtime_error("Loading identity file failed (" + err_string + ")");
97 }

100  class PrivateKeyImpl,
101  class PublicKeyImpl,
102  class SubKeyImpl
103 >
104 void
105 dotsig::OpenPGP::Identity<PrivateKeyImpl, PublicKeyImpl, SubKeyImpl>::Export(
106  const std::string& filename,
107  const std::string& passphrase
108 ) const {
109  std::filesystem::directory_entry entry{filename};
110  if (entry.exists())
111  throw std::runtime_error("Error: File overwrite not yet supported.");
112 
113  Botan::AutoSeeded_RNG rng;
114  auto priv_bytes = Botan::PKCS8::BER_encode(*m_private_key, rng, passphrase);
115  auto pub_bytes = Botan::X509::PEM_encode(*m_public_key); // expects pubkey
116 
117  std::ofstream out_priv(filename),
118  out_pub(filename + ".pub");
119 
120  out_priv << std::string(priv_bytes.begin(), priv_bytes.end()) << std::flush;
121  out_pub << pub_bytes << std::flush;
122 
123  out_priv.close();
124  out_pub.close();
125 }

128  class PrivateKeyImpl,
129  class PublicKeyImpl,
130  class SubKeyImpl
131 >
132 std::string
133 dotsig::OpenPGP::Identity<PrivateKeyImpl, PublicKeyImpl, SubKeyImpl>::Sign(
134  const std::string& message,
135  const std::string& sig_file
136 ) const {
137  Botan::AutoSeeded_RNG rng;
138 
139  // initialize a signer instance with the message
140  Botan::PK_Signer signer(*m_private_key, rng, m_scheme);
141  signer.update(message);
142  std::vector<uint8_t> sig = signer.signature(rng);
143 
144  // saves the signature bytes into a .sig file
145  std::ofstream sig_ptr(sig_file);
146  sig_ptr << std::string(sig.begin(), sig.end());
147  sig_ptr.close();
148 
149  // returns hexadecimal signature notation
150  return Botan::hex_encode(sig);
151 }

157 >
158 bool
159 dotsig::OpenPGP::Identity<PrivateKeyImpl, PublicKeyImpl, SubKeyImpl>::Verify(
160  const std::string& signature,
161  const std::string& message
162 ) const {
163  // initialize a verifier instance with the message
164  Botan::PK_Verifier verifier(*m_public_key, m_scheme);
165  verifier.update(message);
166 
167  std::vector<uint8_t> raw_signature(signature.begin(), signature.end());
168  return verifier.check_signature(raw_signature);
169 }
