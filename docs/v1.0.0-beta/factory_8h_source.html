17  /// \brief A class that implements the factory pattern for identities.
18  ///
19  /// This (minimal) factory pattern implementation deals with IIdentity objects
20  /// and a uses a **pointer creation function** to create pointers of Identity
21  /// classes that are supported by the software. These creator functions may or
22  /// may not prepare the object before it is returned, as deemed necessary.
23  class Factory {
24  /// \brief The factories registry as a map.
25  ///
26  /// Keys are supported algorithm names, e.g. "ecdsa" or "pkcs", and values
27  /// are *function pointers* that shall execute the *pointer creation* for
28  /// the provided class templates based on IIdentity.
29  /// \see Register
30  std::map<std::string, std::function<IIdentity* ()>> m_factories{};
31 
32  public:
33  /// \brief Registers a new factory for an IIdentity derived class.
34  /// \param id The algorithm name used in the association.
35  /// \param maker_fn The *pointer creation function* which returns an IIdentity.
36  /// \return True if the insertion of a new factory succeeded, false otherwise.
37  bool Register(
38  const std::string& id,
39  std::function<IIdentity* ()> maker_fn
40  );
41 
42  /// \brief Creates a new identity for algorithm \a id.
43  ///
44  /// This method uses the factories registry to find the associated pointer

46  /// create the IIdentity instance that is returned.
47  /// \param id The algorithm name used in the association.
48  /// \return The created IIdentity-derived class object or 0 to mark an error.
49  IIdentity* MakeIdentity(const std::string& id);
