44 const dotsig::PKCS::ParentType& dotsig::PKCS::Identity::Import(
45  const std::string& filename,
46  const std::string& passphrase
47 ) {
48  std::string err_string;
49 
50  // try to import an unencrypted public-key (from: id_rsa.pub, id_ecdsa.pub)
51  try {
52  Botan::DataSource_Stream input(filename);
53  std::unique_ptr<Botan::Public_Key> pub = Botan::X509::load_key(input);
54 
55  m_public_key = std::make_unique<dotsig::PKCS::PublicKey>(
56  pub->algorithm_identifier(),
57  pub->public_key_bits()
58  );
59 
60  return *this;
61  }
62  catch(Botan::Exception& e) {}
63 
64  // or try to load using an encrypted private key (from: id_rsa, id_ecdsa)
65  try {
66  Botan::DataSource_Stream input(filename);
67  std::unique_ptr<Botan::Private_Key> priv = Botan::PKCS8::load_key(
68  input,
69  passphrase
70  );
71 
72  m_private_key = std::make_unique<dotsig::PKCS::PrivateKey>(
73  priv->algorithm_identifier(), priv->private_key_bits()
74  );
75 
76  m_public_key = std::make_unique<dotsig::PKCS::PublicKey>(
77  m_private_key->algorithm_identifier(),
78  m_private_key->public_key_bits()
79  );
80 
81  return *this;
82  }
83  catch(Botan::Exception& e) {
84  err_string = e.what();
85  }
86 
87  // private key could not be loaded
88  throw std::runtime_error("Loading identity file failed (" + err_string + ")");
89 }

95  std::filesystem::directory_entry entry{filename};

98 
99  Botan::AutoSeeded_RNG rng;
100  auto priv_bytes = Botan::PKCS8::BER_encode(*m_private_key, rng, passphrase);
101  auto pub_bytes = Botan::X509::PEM_encode(*m_public_key);
102 
103  std::ofstream out_priv(filename),
104  out_pub(filename + ".pub");
105 
106  out_priv << std::string(priv_bytes.begin(), priv_bytes.end()) << std::flush;
107  out_pub << pub_bytes << std::flush;
108 
109  out_priv.close();
110  out_pub.close();
111 }

113 std::string dotsig::PKCS::Identity::Sign(
114  const std::string& message,
115  const std::string& sig_file
116 ) const {
117  Botan::AutoSeeded_RNG rng;
118 
119  // initialize a signer instance with the message
120  Botan::PK_Signer signer(*m_private_key, rng, "SHA-256");
121  signer.update(message);
122  std::vector<uint8_t> sig = signer.signature(rng);
123 
124  // saves the signature bytes into a .sig file
125  std::ofstream sig_ptr(sig_file);
126  sig_ptr << std::string(sig.begin(), sig.end());
127  sig_ptr.close();
128 
129  // returns hexadecimal signature notation
130  return Botan::hex_encode(sig);
131 }
